## Render script (the entire active code, lines 37–95 of index.html)

```javascript
(function renderScene() {
  var canvas = document.getElementById('game');
  if (!canvas || !canvas.getContext) return;
  var ctx = canvas.getContext('2d');
  if (!ctx) return;

  // 1. Clear (background = walls)
  ctx.fillStyle = '#000';
  ctx.fillRect(0, 0, 640, 480);

  // 2. Floor: cols 1..18, rows 1..13, tile 32x32
  ctx.fillStyle = '#333';
  for (var row = 1; row <= 13; row++) {
    for (var col = 1; col <= 18; col++) {
      ctx.fillRect(col * 32, row * 32, 32, 32);
    }
  }

  // 3. Walls: edges (col 0, col 19, row 0, row 14) — same color as bg
  ctx.fillStyle = '#000';
  for (var c = 0; c < 20; c++) {
    ctx.fillRect(c * 32, 0 * 32, 32, 32);
    ctx.fillRect(c * 32, 14 * 32, 32, 32);
  }
  for (var r = 1; r <= 13; r++) {
    ctx.fillRect(0 * 32, r * 32, 32, 32);
    ctx.fillRect(19 * 32, r * 32, 32, 32);
  }

  // 4. Player: blue square at tile (2, 2)
  ctx.fillStyle = 'blue';
  ctx.fillRect(2 * 32, 2 * 32, 32, 32);

  // 5. Slime: red/purple square at tile (17, 12)
  ctx.fillStyle = 'rgb(180, 40, 180)';
  ctx.fillRect(17 * 32, 12 * 32, 32, 32);

  // 6–9. HUD + state text + controls placeholder
  ctx.fillStyle = '#fff';
  ctx.font = 'bold 16px monospace';
  ctx.textBaseline = 'top';
  ctx.fillText('Player HP: 5', 8, 8);
  ctx.fillText('Slime HP: 3', 640 - 130, 8);
  ctx.fillText('DUNGEON SPARK', (640 / 2) - 60, (480 / 2) - 8);
  ctx.fillText('WASD/Arrows: move | Space: attack | R: restart', 80, 480 - 24);
})();
```
