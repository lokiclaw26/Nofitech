```js
// Stage 4: slime AI — greedy Manhattan chase.
// Returns true if (c, r) is interior floor AND not the player's tile.
// (Player tile is "free" in the sense of "slime cannot stand on it"
// for V1 — no contact damage yet.)
function inBoundsAndFree(c, r) {
  if (c < COL_MIN || c > COL_MAX) return false; // wall: left/right edges
  if (r < ROW_MIN || r > ROW_MAX) return false; // wall: top/bottom edges
  if (c === player.col && r === player.row) return false; // player tile blocks
  return true;
}

// Slime tick: pick a cardinal direction that reduces Manhattan distance
// to the player, 1 tile, walls + player tile block. NO damage.
function slimeTick() {
  var dx = player.col - slime.col;
  var dy = player.row - slime.row;
  var candidates = [];
  if (dx > 0 && inBoundsAndFree(slime.col + 1, slime.row)) candidates.push({dc: 1,  dr: 0,  dir: 'right'});
  if (dx < 0 && inBoundsAndFree(slime.col - 1, slime.row)) candidates.push({dc: -1, dr: 0,  dir: 'left' });
  if (dy > 0 && inBoundsAndFree(slime.col, slime.row + 1)) candidates.push({dc: 0,  dr: 1,  dir: 'down' });
  if (dy < 0 && inBoundsAndFree(slime.col, slime.row - 1)) candidates.push({dc: 0,  dr: -1, dir: 'up'   });
  if (candidates.length === 0) return; // blocked — slime waits this tick
  // Tie-break: up(0) > right(1) > down(2) > left(3) per technical-values.md
  var order = { up: 0, right: 1, down: 2, left: 3 };
  candidates.sort(function (a, b) { return order[a.dir] - order[b.dir]; });
  var pick = candidates[0];
  slime.col += pick.dc;
  slime.row += pick.dr;
  render();
}

// Initial draw
render();

// Stage 4: start the slime AI — 500ms tick, 1 tile per tick, no damage.
setInterval(slimeTick, 500);
```
