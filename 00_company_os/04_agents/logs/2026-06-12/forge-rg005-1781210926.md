```js
// Stage 5: facing-string -> unit vector. Safe default (0,0) so a
// mis-set facing string never produces a wild swing.
function facingToDir(f) {
  if (f === 'right') return { dx:  1, dy:  0 };
  if (f === 'left')  return { dx: -1, dy:  0 };
  if (f === 'down')  return { dx:  0, dy:  1 };
  if (f === 'up')    return { dx:  0, dy: -1 };
  return { dx: 0, dy: 0 };
}

// Stage 5: player swing. 1 tile in facing direction, 1 damage to the
// slime if it's there, 400ms cooldown, yellow flash on the target
// tile for the cooldown duration. No contact damage, no win state.
function playerAttack() {
  var now = Date.now();
  if (now - lastAttackTime < 400) return; // cooldown guard
  lastAttackTime = now;
  var d  = facingToDir(player.facing);
  var tc = player.col + d.dx;
  var tr = player.row + d.dy;
  if (tc < COL_MIN || tc > COL_MAX || tr < ROW_MIN || tr > ROW_MAX) {
    return; // wall: no flash, no damage
  }
  attackFlash = { col: tc, row: tr, until: now + 400 };
  if (tc === slime.col && tr === slime.row) {
    slime.hp -= 1; // can go to 0; no win state this stage
  }
  render();
}

// In the existing keydown handler:
} else if (e.key === ' ') {
  e.preventDefault();
  playerAttack();
}
```
