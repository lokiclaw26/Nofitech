    warnings_list = []   # ALL warn-level entries, no cap

    ...

        entry = {

            "level": level,
            "title": meta.get("title") or f.stem,
        }
        if level == "error":
            errors_list.append(entry)
        elif level == "warn":
            warnings_list.append(entry)
        else:
            info_list.append(entry)

    warnings_list.sort(key=lambda e: e["ts"], reverse=True)

  const evWarnsErrs = ev.filter(e => e.level === "warn" || e.level === "error");

  const extras = evWarnsErrs.filter(e => !seenSrcTs.has((e.source||"") + "::" + (e.ts||"")));

  if (totalWarns === 0 && totalErrs === 0 && blockedTasks === 0) {
    .innerHTML =
      `<div class="action-row ok"><span class="icon">✓</span> No active warnings — all systems nominal.</div>`;

  }
  ...
  for (const w of warns) rows.push(_warningsRow("warn", w, w.title || w.message || "—"));
